    def Nat.any (f : Nat → Bool) :
    Nat → Bool

    any f n = true iff there is i in [0, n-1] s.t. f i = true

    Equations
    • Nat.any f 0 = false
    • Nat.any f (Nat.succ n) = (Nat.any f n || f n)

    @[specialize #[]]
    def Nat.all (f : Nat → Bool) :
    Nat → Bool

    all f n = true iff every i in [0, n-1] satisfies f i = true

    Equations
    • Nat.all f 0 = true
    • Nat.all f (Nat.succ n) = (Nat.all f n && f n)
